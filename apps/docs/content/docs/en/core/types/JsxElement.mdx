## Overview

`JsxElement` represents structured HTML-like elements within translatable JSX content. It defines the shape of elements that can contain children, attributes, and metadata while maintaining translatability across languages.

```typescript
type JsxElement = {
  t?: string;      // tag name
  i?: number;      // id
  d?: GTProp;      // GT data/properties
  c?: JsxChildren; // children
};
```

This type enables creation of complex, nested content structures that preserve HTML semantics and accessibility features during translation.

## Type Definition

### Properties

| Property | Type | Description |
|----------|------|-------------|
| `t?` | `string` | HTML tag name (e.g., 'div', 'span', 'a', 'button') |
| `i?` | `number` | Optional internal identifier for the element |
| `d?` | `GTProp` | GT properties containing HTML attributes and metadata |
| `c?` | `JsxChildren` | Child content (text, variables, or nested elements) |

### GTProp Structure

```typescript
type GTProp = {
  b?: Record<string, JsxChildren>; // Branches for conditional content
  t?: 'p' | 'b';                   // Branch transformation type
} & HtmlContentPropKeysRecord;     // HTML attributes
```

### Supported HTML Attributes (via GTProp)

| GT Key | HTML Attribute | Description |
|--------|----------------|-------------|
| `pl` | `placeholder` | Input placeholder text |
| `ti` | `title` | Tooltip/title attribute |
| `alt` | `alt` | Alternative text for images |
| `arl` | `aria-label` | Accessibility label |
| `arb` | `aria-labelledby` | References to labelling elements |
| `ard` | `aria-describedby` | References to describing elements |

## Example

### Basic Elements

```typescript copy
import { JsxElement, JsxChildren } from 'generaltranslation';

// Simple div element
const simpleDiv: JsxElement = {
  t: 'div',
  c: ['Hello World']
};

// Paragraph with ID
const paragraph: JsxElement = {
  t: 'p',
  i: 1,
  c: ['This is a paragraph with an internal ID']
};

// Span with attributes
const spanWithTitle: JsxElement = {
  t: 'span',
  d: {
    ti: 'Additional information'  // title attribute
  },
  c: ['Hover for tooltip']
};
```

  },
  c: [
    '$',
    { k: 'price', v: 'n' } as Variable
  ]

### Helper Functions

```typescript copy
// Helper to create basic elements
function createElement(
  tag: string,
  children?: JsxChildren,
  props?: GTProp,
  id?: number
): JsxElement {
  const element: JsxElement = { t: tag };
  
  if (children !== undefined) {
    element.c = children;
  }
  
  if (props) {
    element.d = props;
  }
  
  if (id !== undefined) {
    element.i = id;
  }
  
  return element;
}

// Helper for elements with just text
function textElement(tag: string, text: string, props?: GTProp): JsxElement {
  return createElement(tag, [text], props);
}

// Helper for elements with variables
function variableElement(
  tag: string,
  variableKey: string,
  variableType?: VariableType,
  props?: GTProp
): JsxElement {
  const variable: Variable = { k: variableKey };
  if (variableType) variable.v = variableType;
  
  return createElement(tag, [variable], props);
}

// Usage examples
const helpers = {
  title: textElement('h1', 'Page Title'),
  price: variableElement('span', 'productPrice', 'c', { ti: 'Current price' }),
  description: createElement('p', [
    'Product by ',
    { k: 'brandName' } as Variable
  ])
};
```

### Element Validation

```typescript copy
function isValidJsxElement(obj: any): obj is JsxElement {
  if (typeof obj !== 'object' || obj === null) {
    return false;
  }
  
  // Must have at least one property
  const hasProperties = Object.keys(obj).length > 0;
  if (!hasProperties) return false;
  
  // Check optional properties
  if (obj.t !== undefined && typeof obj.t !== 'string') return false;
  if (obj.i !== undefined && typeof obj.i !== 'number') return false;
  if (obj.d !== undefined && typeof obj.d !== 'object') return false;
  // Note: c validation would require recursive JsxChildren validation
  
  return true;
}

function createSafeElement(
  tag: string,
  children?: JsxChildren,
  props?: GTProp
): JsxElement {
  const element = { t: tag, c: children, d: props };
  
  if (!isValidJsxElement(element)) {
    throw new Error('Invalid JsxElement structure');
  }
  
  return element;
}
```

## Notes

* All properties are optional, allowing for flexible element creation
* The `t` property should contain valid HTML tag names
* GT properties (`d`) use shortened keys to reduce payload size
* Accessibility attributes are fully supported through GT properties  
* Elements can be nested to arbitrary depth while maintaining translatability
* The `i` property is used internally for optimization and should be unique within a content tree
* Children can mix text, variables, and other elements freely
