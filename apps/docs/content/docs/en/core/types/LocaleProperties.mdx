## Overview

`LocaleProperties` is a comprehensive interface that provides detailed linguistic and regional information about a locale. It includes display names, native names, region/script details, and emoji representations for any valid BCP-47 locale code.

```typescript
interface LocaleProperties {
  code: string;
  name: string;
  nativeName: string;
  languageCode: string;
  languageName: string;
  nativeLanguageName: string;
  nameWithRegionCode: string;
  nativeNameWithRegionCode: string;
  regionCode: string;
  regionName: string;
  nativeRegionName: string;
  scriptCode: string;
  scriptName: string;
  nativeScriptName: string;
  maximizedCode: string;
  maximizedName: string;
  nativeMaximizedName: string;
  minimizedCode: string;
  minimizedName: string;
  nativeMinimizedName: string;
  emoji: string;
}
```

This interface is returned by `getLocaleProperties()` methods and provides everything needed for rich locale-aware user interfaces.

<Callout>
**Maximization & Minimization:**
The interface includes both maximized (with likely script/region) and minimized (shortest valid) locale codes, following Unicode CLDR standards for locale processing.
</Callout>

### Building Rich Locale Selector

```typescript
interface LocaleOption {
  code: string;
  displayName: string;
  nativeName: string;
  flag: string;
  region: string;
  isRTL: boolean;
}

function createRichLocaleSelector(
  gt: GT,
  locales: string[]
): LocaleOption[] {
  return locales.map(locale => {
    const props = gt.getLocaleProperties(locale);
    
    return {
      code: locale,
      displayName: props.name,
      nativeName: props.nativeName,
      flag: props.emoji,
      region: props.regionName,
      isRTL: gt.getLocaleDirection(locale) === 'rtl'
    };
  });
}

const options = createRichLocaleSelector(gt, [
  'ar-SA',  // Arabic (Saudi Arabia)
  'en-US',  // English (United States)
  'zh-CN',  // Chinese (China)
  'he-IL'   // Hebrew (Israel)
]);

// Results include RTL information, flags, and rich naming
```

### Locale Hierarchy Visualization

```typescript
interface LocaleHierarchy {
  language: {
    code: string;
    name: string;
    nativeName: string;
  };
  regions: Array<{
    code: string;
    name: string;
    nativeName: string;
    emoji: string;
  }>;
}

function createLocaleHierarchy(
  gt: GT,
  locales: string[]
): LocaleHierarchy[] {
  const grouped = new Map<string, LocaleProperties[]>();
  
  // Group by language code
  locales.forEach(locale => {
    const props = gt.getLocaleProperties(locale);
    const langCode = props.languageCode;
    
    if (!grouped.has(langCode)) {
      grouped.set(langCode, []);
    }
    grouped.get(langCode)!.push(props);
  });
  
  return Array.from(grouped.entries()).map(([langCode, regionProps]) => ({
    language: {
      code: langCode,
      name: regionProps[0].languageName,
      nativeName: regionProps[0].nativeLanguageName
    },
    regions: regionProps.map(props => ({
      code: props.regionCode,
      name: props.regionName,
      nativeName: props.nativeRegionName,
      emoji: props.emoji
    }))
  }));
}
```

## Advanced Usage Patterns

### Locale Comparison Tool

```typescript
function compareLocales(gt: GT, locale1: string, locale2: string) {
  const props1 = gt.getLocaleProperties(locale1);
  const props2 = gt.getLocaleProperties(locale2);
  
  return {
    sameLanguage: props1.languageCode === props2.languageCode,
    sameScript: props1.scriptCode === props2.scriptCode,
    sameRegion: props1.regionCode === props2.regionCode,
    similarity: calculateSimilarity(props1, props2),
    comparison: {
      [locale1]: {
        display: props1.name,
        native: props1.nativeName,
        minimized: props1.minimizedCode
      },
      [locale2]: {
        display: props2.name,  
        native: props2.nativeName,
        minimized: props2.minimizedCode
      }
    }
  };
}

function calculateSimilarity(props1: LocaleProperties, props2: LocaleProperties): number {
  let score = 0;
  if (props1.languageCode === props2.languageCode) score += 50;
  if (props1.scriptCode === props2.scriptCode) score += 30;
  if (props1.regionCode === props2.regionCode) score += 20;
  return score;
}

const comparison = compareLocales(gt, 'en-US', 'en-GB');
// { sameLanguage: true, sameScript: true, sameRegion: false, similarity: 80, ... }
```

### Smart Locale Fallback

```typescript
function findBestFallback(
  gt: GT,
  requestedLocale: string,
  availableLocales: string[]
): { locale: string; score: number; reason: string } {
  const requestedProps = gt.getLocaleProperties(requestedLocale);
  
  const candidates = availableLocales.map(locale => {
    const props = gt.getLocaleProperties(locale);
    let score = 0;
    let reasons: string[] = [];
    
    if (props.languageCode === requestedProps.languageCode) {
      score += 50;
      reasons.push('same language');
    }
    
    if (props.scriptCode === requestedProps.scriptCode) {
      score += 30;
      reasons.push('same script');
    }
    
    if (props.regionCode === requestedProps.regionCode) {
      score += 20;
      reasons.push('same region');
    }
    
    return {
      locale,
      score,
      reason: reasons.join(', ') || 'different language family'
    };
  });
  
  return candidates.sort((a, b) => b.score - a.score)[0];
}

const fallback = findBestFallback(gt, 'fr-CA', ['fr-FR', 'en-CA', 'es-ES']);
// { locale: 'fr-FR', score: 80, reason: 'same language, same script' }
```

## Notes

- All names are localized according to the GT instance's source locale
- Script and region codes follow Unicode CLDR standards
- Maximization/minimization follows Unicode locale algorithms
- Native names use the locale's own writing system
- Emoji selection prioritizes region over language for visual representation
- Properties are computed dynamically and may vary based on Intl API availability
