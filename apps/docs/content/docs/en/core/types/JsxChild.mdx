## Overview

`JsxChild` represents a single element that can appear within JSX content. It's a union type that encompasses all possible individual items that can be part of translatable JSX structure.

```typescript
type JsxChild = string | JsxElement | Variable;
```

This type is the building block for [`JsxChildren`](/docs/core/types/JsxChildren), allowing flexible composition of text, structured elements, and dynamic variables in translatable content.

## Type Definition

### Union Members

| Type | Description | Example |
|------|-------------|---------|
| `string` | Plain text content | `"Hello world"`, `"Click here"` |
| `JsxElement` | Structured HTML-like element | `{ t: 'div', c: ['content'] }` |
| `Variable` | Dynamic placeholder for runtime substitution | `{ k: 'userName', v: 'v' }` |

## Example

### String Children

```typescript copy
import { JsxChild } from 'generaltranslation';

// Plain text child
const textChild: JsxChild = "Welcome to our application!";

// Text that will be translated
const greeting: JsxChild = "Hello and welcome!";

// Label text
const buttonLabel: JsxChild = "Submit Form";
```

### Variable Children

```typescript copy
import { JsxChild, Variable } from 'generaltranslation';

// Simple text variable
const nameVar: JsxChild = { k: 'userName' } as Variable;

// Formatted number variable  
const countVar: JsxChild = { k: 'itemCount', v: 'n' } as Variable;

// Currency variable
const priceVar: JsxChild = { k: 'totalPrice', v: 'c' } as Variable;

// Date variable
const dateVar: JsxChild = { k: 'lastLogin', v: 'd' } as Variable;
```

### JsxElement Children

```typescript copy
import { JsxChild, JsxElement } from 'generaltranslation';

// Simple element
const linkElement: JsxChild = {
  t: 'a',
  d: { ti: 'Go to homepage' },
  c: ['Home']
};

// Element with variables
const userProfile: JsxChild = {
  t: 'div',
  c: [
    'User: ',
    { k: 'userName' } as Variable
  ]
};

// Complex nested element
const productCard: JsxChild = {
  t: 'article',
  d: { ti: 'Product information' },
  c: [
    {
      t: 'h3',
      c: [{ k: 'productName' } as Variable]
    },
    {
      t: 'p',
      c: [
        'Price: ',
        { k: 'price', v: 'c' } as Variable
      ]
    }
  ]
};
```

  // Variable child (number)

  // Element child
  {
    t: 'a',

### Helper Functions for Child Creation

```typescript copy
// Type guards for JsxChild
function isStringChild(child: JsxChild): child is string {
  return typeof child === 'string';
}

function isVariableChild(child: JsxChild): child is Variable {
  return typeof child === 'object' && child !== null && 'k' in child;
}

function isElementChild(child: JsxChild): child is JsxElement {
  return typeof child === 'object' && child !== null && 't' in child;
}

// Helper to create different child types
function createChild(type: 'string', content: string): string;
function createChild(type: 'variable', key: string, varType?: VariableType): Variable;
function createChild(type: 'element', tag: string, children: JsxChild[]): JsxElement;
function createChild(type: string, ...args: any[]): JsxChild {
  switch (type) {
    case 'string':
      return args[0];
    case 'variable':
      return { k: args[0], ...(args[1] && { v: args[1] }) };
    case 'element':
      return { t: args[0], c: args[1] };
    default:
      throw new Error(`Unknown child type: ${type}`);
  }
}

// Usage examples
const children = [
  createChild('string', 'Hello'),
  createChild('variable', 'name'),
  createChild('element', 'strong', ['Important!']),
  createChild('variable', 'count', 'n')
];
```

### Conditional Child Creation

```typescript copy
function createConditionalChild(
  condition: boolean,
  trueChild: JsxChild,
  falseChild?: JsxChild
): JsxChild | null {
  if (condition) {
    return trueChild;
  }
  return falseChild || null;
}

// Usage in content creation
const dynamicContent = [
  'Welcome',
  createConditionalChild(
    true,
    { k: 'userName' } as Variable,
    'Guest'
  ),
  '!'
].filter(Boolean) as JsxChild[];
```

### Child Validation

```typescript copy
function validateJsxChild(child: any): child is JsxChild {
  // String validation
  if (typeof child === 'string') {
    return true;
  }
  
  // Variable validation
  if (typeof child === 'object' && child !== null && 'k' in child) {
    return typeof child.k === 'string';
  }
  
  // Element validation
  if (typeof child === 'object' && child !== null && 't' in child) {
    return typeof child.t === 'string';
  }
  
  return false;
}

function validateJsxChildArray(children: any[]): children is JsxChild[] {
  return children.every(validateJsxChild);
}

// Usage
const potentialChildren = [
  'Hello',
  { k: 'name' },
  { t: 'span', c: ['world'] }
];

if (validateJsxChildArray(potentialChildren)) {
  // TypeScript now knows these are valid JsxChild[]
  console.log('All children are valid');
}
```

## Notes

* `JsxChild` is the singular form, while `JsxChildren` can be either a single child or array of children
* String children are the simplest form and represent plain text content
* Variable children enable dynamic content substitution at runtime
* Element children provide structure and can contain nested children
* All child types can be mixed freely within the same content structure
* Each child type maintains its own properties while fitting into the unified type system
